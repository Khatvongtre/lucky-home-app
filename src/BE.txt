using LuckyHome.Api.Data;
using LuckyHome.Api.Models;
using Microsoft.AspNetCore.Authorization;
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;
using System.Security.Claims;

namespace LuckyHome.Api.Controllers
{
    [Authorize]
    [Route("api/funds")]
    [ApiController]
    public class FundController : ControllerBase
    {
        private readonly LuckyHomeDbContext _db;

        public FundController(LuckyHomeDbContext db)
        {
            _db = db;
        }

        private string GetUserId() => User.FindFirstValue(ClaimTypes.NameIdentifier);

        // 1. GET: Lấy danh sách Hũ của user
        [HttpGet]
        public async Task<IActionResult> GetFunds()
        {
            var userId = GetUserId();
            if (string.IsNullOrEmpty(userId)) return Unauthorized();

            var funds = await _db.Funds.Where(j => j.UserId == userId).ToListAsync();

            // Nếu chưa có hũ nào, tự động tạo 6 hũ mặc định
            if (!funds.Any())
            {
                funds = new List<Fund>
                {
                    new Fund { Id = 1, UserId = userId, Name = "Thiết yếu", Percent = 55, Balance = 0, Color = "bg-blue-500", Text = "text-blue-600", Bg = "bg-blue-50", IconName = "Wallet" },
                    new Fund { Id = 2, UserId = userId, Name = "Tiết kiệm", Percent = 10, Balance = 0, Color = "bg-emerald-500", Text = "text-emerald-600", Bg = "bg-emerald-50", IconName = "TrendingUp" },
                    new Fund { Id = 3, UserId = userId, Name = "Đầu tư", Percent = 10, Balance = 0, Color = "bg-purple-500", Text = "text-purple-600", Bg = "bg-purple-50", IconName = "PieChart" },
                    new Fund { Id = 4, UserId = userId, Name = "Giáo dục", Percent = 10, Balance = 0, Color = "bg-amber-500", Text = "text-amber-600", Bg = "bg-amber-50", IconName = "GraduationCap" },
                    new Fund { Id = 5, UserId = userId, Name = "Hưởng thụ", Percent = 10, Balance = 0, Color = "bg-pink-500", Text = "text-pink-600", Bg = "bg-pink-50", IconName = "Plane" },
                    new Fund { Id = 6, UserId = userId, Name = "Cho đi", Percent = 5, Balance = 0, Color = "bg-rose-500", Text = "text-rose-600", Bg = "bg-rose-50", IconName = "HeartHandshake" }
                };
                _db.Funds.AddRange(funds);
                await _db.SaveChangesAsync();
            }

            return Ok(funds.OrderBy(j => j.Id));
        }

        // 2. PUT: Cập nhật thiết lập các Hũ (Thêm, Xóa, Đổi %, Đổi tên)
        [HttpPut]
        public async Task<IActionResult> UpdateFunds([FromBody] List<Fund> updatedFunds)
        {
            var userId = GetUserId();
            var existingFunds = await _db.Funds.Where(j => j.UserId == userId).ToListAsync();

            // Xóa các hũ mà người dùng đã xóa trên giao diện
            var fundsToRemove = existingFunds.Where(ej => !updatedFunds.Any(uj => uj.Id == ej.Id)).ToList();
            if (fundsToRemove.Any()) _db.Funds.RemoveRange(fundsToRemove);

            foreach (var fund in updatedFunds)
            {
                var existing = existingFunds.FirstOrDefault(j => j.Id == fund.Id);
                if (existing != null)
                {
                    // Chỉ cập nhật giao diện và %, KHÔNG ghi đè số dư (Balance)
                    existing.Name = fund.Name;
                    existing.Percent = fund.Percent;
                    existing.Color = fund.Color;
                    existing.Text = fund.Text;
                    existing.Bg = fund.Bg;
                    existing.IconName = fund.IconName;
                }
                else
                {
                    fund.UserId = userId;
                    // Chấp nhận jar.Balance = 0 từ FE
                    _db.Funds.Add(fund);
                }
            }

            await _db.SaveChangesAsync();
            return Ok(await _db.Funds.Where(j => j.UserId == userId).OrderBy(j => j.Id).ToListAsync());
        }

        // 3. GET: Lấy danh sách giao dịch
        [HttpGet("transactions")]
        public async Task<IActionResult> GetTransactions()
        {
            var userId = GetUserId();
            var txs = await _db.FundTransactions
                .Where(t => t.UserId == userId)
                .OrderByDescending(t => t.Date)
                .ToListAsync();
            return Ok(txs);
        }

        // 4. POST: Ghi nhận giao dịch (Nạp/Chi)
        [HttpPost("transactions")]
        public async Task<IActionResult> AddTransactions([FromBody] List<FundTransaction> transactions)
        {
            var userId = GetUserId();
            var funds = await _db.Funds.Where(j => j.UserId == userId).ToListAsync();

            foreach (var tx in transactions)
            {
                tx.UserId = userId;
                if (tx.Id == Guid.Empty) tx.Id = Guid.NewGuid();
                if (tx.Date == default) tx.Date = DateTime.Now;

                var fund = funds.FirstOrDefault(j => j.Id == tx.FundId);
                if (fund != null)
                {
                    // Cộng/Trừ số tiền trực tiếp vào hũ
                    if (tx.Type == "in") fund.Balance += tx.Amount;
                    else if (tx.Type == "out") fund.Balance -= tx.Amount;
                }
                _db.FundTransactions.Add(tx);
            }

            await _db.SaveChangesAsync();
            return Ok(new { success = true });
        }

        // 5. DELETE: Xóa giao dịch và hoàn tiền lại vào Hũ
        [HttpDelete("transactions/{id}")]
        public async Task<IActionResult> DeleteTransaction(Guid id)
        {
            var userId = GetUserId();
            var tx = await _db.FundTransactions.FirstOrDefaultAsync(t => t.Id == id && t.UserId == userId);
            if (tx == null) return NotFound("Không tìm thấy giao dịch.");

            var fund = await _db.Funds.FirstOrDefaultAsync(j => j.Id == tx.FundId && j.UserId == userId);
            if (fund != null)
            {
                // Hoàn lại số dư vào hũ
                if (tx.Type == "in") fund.Balance -= tx.Amount;
                else if (tx.Type == "out") fund.Balance += tx.Amount;
            }

            _db.FundTransactions.Remove(tx);
            await _db.SaveChangesAsync();
            return Ok(new { success = true });
        }
    }
}
